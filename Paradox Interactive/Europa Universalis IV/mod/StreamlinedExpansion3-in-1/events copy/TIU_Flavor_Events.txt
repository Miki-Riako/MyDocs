###Flavor Events for Gensokyo###

namespace = flavor_tiu

#Strawberries in gensokyo
country_event = {
    id = flavor_tiu.1
    title = flavor_tiu.1.t
    desc = flavor_tiu.1.d
    picture = TIU_qtstrawberries

	fire_only_once = yes
	major = yes

    trigger = {
        OR = {
            has_global_flag = trade_goods_expanded_mod_active
            has_global_flag = trade_goods_exapanded_mod_active
        }
        capital_scope = {
            continent = gensokyo_continent
        }
        any_owned_province = {
            NOT = { continent = gensokyo_continent }
            trade_goods = strawberry
        }
    }
    
    mean_time_to_happen = {
        days = 1
    }
    
    option = {
        name = flavor_tiu.1.a
        add_treasury = 500
        custom_tooltip = genso_berries.tt
        hidden_effect = {
            random_owned_province = {
                limit = {
                    NOT = { trade_goods = strawberry }
                }
                change_trade_goods = strawberry
            }
            random_owned_province = {
                limit = {
                    NOT = { trade_goods = strawberry }
                }
                change_trade_goods = strawberry
            }
            random_owned_province = {
                limit = {
                    NOT = { trade_goods = strawberry }
                }
                change_trade_goods = strawberry
            }
            5084 = { change_trade_goods = strawberry }
            5082 = { change_trade_goods = strawberry }
            4996 = { change_trade_goods = strawberry }
        }
		change_price = {
			trade_goods = strawberry
			key = STRAWBERRIES
			value = 0.20
			duration = 7300
        }
    }
}
