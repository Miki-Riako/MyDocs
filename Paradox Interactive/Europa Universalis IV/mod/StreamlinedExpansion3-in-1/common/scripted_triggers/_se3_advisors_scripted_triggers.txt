se3_has_adm_advisor_with_skill = {
	OR = {
		treasurer = $skill$
		philosopher = $skill$
		artist = $skill$
		theologian = $skill$
		master_of_mint = $skill$
		inquisitor = $skill$
		natural_scientist = $skill$
		# administrator = $skill$
		# state_overseer = $skill$
		# civil_engineer = $skill$
		# researcher = $skill$
		# reformist = $skill$
		# religious_representative = $skill$
		# julius_archbishop = $skill$
		# fu_accountant = $skill$
		# fu_royal_imam = $skill$
		# fu_city_planner = $skill$
		# fu_cartographer = $skill$
		# fu_head_of_patriarchate = $skill$
		# fu_right_hand_captain = $skill$
		if = { #TOUHOU
			limit = { is_mod_active = "Touhou Universalis Updated (CN)"}
			OR = {
				th_advisor_flandre = $skill$
				th_advisor_eirin = $skill$
				th_advisor_keine = $skill$
				th_advisor_patchouli = $skill$
				th_advisor_daiyousei = $skill$
				th_advisor_daiyousei_upgraded = $skill$
				th_advisor_hecatia = $skill$
				th_advisor_sagume = $skill$
			}
		}
		if = { #Maid
			limit = { is_mod_active = "女仆之国"}
			OR = {
				ms_advisor_scientist = $skill$
				ms_advisor_green = $skill$
				ms_advisor_blue = $skill$
				ms_advisor_grey = $skill$
			}
		}
	}
}
