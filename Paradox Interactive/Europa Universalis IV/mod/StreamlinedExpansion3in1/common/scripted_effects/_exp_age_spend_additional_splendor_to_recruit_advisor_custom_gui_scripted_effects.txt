# Vanilla
ase_spend_additional_splendor_to_recruit_advisor_entry_1_effect = {
	add_splendor = -400
	if = {
		limit = {
			has_country_flag = ase_spend_additional_splendor_recruit_advisor_adm_opened
		}
		define_advisor = {
			type = philosopher
			culture = ROOT
			cost_multiplier = 0.75
			skill = 2
		}
	}
	else_if = {
		limit = {
			has_country_flag = ase_spend_additional_splendor_recruit_advisor_dip_opened
		}
		define_advisor = {
			type = statesman
			culture = ROOT
			cost_multiplier = 0.75
			skill = 2
		}
	}
	else_if = {
		limit = {
			has_country_flag = ase_spend_additional_splendor_recruit_advisor_mil_opened
		}
		define_advisor = {
			type = army_reformer
			culture = ROOT
			cost_multiplier = 0.75
			skill = 2
		}
	}
}
